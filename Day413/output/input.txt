#include<iostream>
#include<cstdio>
using namespace std;

int f(int x)
{
    if(x == 1)
    {
        return 1;
    }else
    {
        if(x == 2)
            return 1;
        else
            return f(x - 1) + f(x - 2);
    }
}

int main()
{
    int i;

    i = 2;

    cout << f(i) << endl;
    return 0;
}
